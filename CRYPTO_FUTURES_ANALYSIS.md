def get_binance_futures_candles(symbol='BTCUSDT', interval='1h', limit=1000):
    """
    Get historical futures candles from Binance
    
    Args:
        symbol: Trading pair (e.g., 'BTCUSDT', 'ETHUSDT')

        candles = response.json()
